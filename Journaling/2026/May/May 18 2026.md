1import React, { useState, useEffect, useRef } from "react";

2

3const STEPS = 16;

4const TRACKS = [

5  { id: "kick", label: "Kick", freq: 150 },

6  { id: "snare", label: "Snare", freq: 250 },

7  { id: "hihat", label: "Hi-Hat", freq: 800 },

8  { id: "perc", label: "Perc", freq: 400 },

9];

10

11const COLORS = {

12  kick: "#ff0055",

13  snare: "#00f2ff",

14  hihat: "#bcff00",

15  perc: "#ffaa00",

16};

17

18export default function NeonStepSequencer() {

19  const [isPlaying, setIsPlaying] = useState(false);

20  const [currentStep, setCurrentStep] = useState(0);

21  const [bpm, setBpm] = useState(120);

22  const [grid, setGrid] = useState<Record<string, boolean[]>>(() => {

23    const initialGrid: Record<string, boolean[]> = {};

24    TRACKS.forEach((track) => {

25      initialGrid[track.id] = new Array(STEPS).fill(false);

26    });

27    return initialGrid;

28  });

29

30  const audioCtx = useRef<AudioContext | null>(null);

31  const timerId = useRef<number | null>(null);

32  const stepRef = useRef(0);

33

34  const playSound = (freq: number) => {

35    if (!audioCtx.current) return;

36    const osc = audioCtx.current.createOscillator();

37    const gain = audioCtx.current.createGain();

38    

39    osc.type = "sine";

40    osc.frequency.setValueAtTime(freq, audioCtx.current.currentTime);

41    osc.frequency.exponentialRampToValueAtTime(0.01, audioCtx.current.currentTime + 0.1);

42    

43    gain.gain.setValueAtTime(0.3, audioCtx.current.currentTime);

44    gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.current.currentTime + 0.1);

45    

46    osc.connect(gain);

47    gain.connect(audioCtx.current.destination);

48    

49    osc.start();

50    osc.stop(audioCtx.current.currentTime + 0.1);

51  };

52

53  useEffect(() => {

54    if (isPlaying) {

55      const interval = (60 / bpm / 4) * 1000;

56      timerId.current = window.setInterval(() => {

57        setCurrentStep((prev) => {

58          const next = (prev + 1) % STEPS;

59          TRACKS.forEach((track) => {

60            if (grid[track.id][next]) {

61              playSound(track.freq);

62            }

63          });

64          return next;

65        });

66      }, interval);

67    } else {

68      if (timerId.current) clearInterval(timerId.current);

69    }

70    return () => {

71      if (timerId.current) clearInterval(timerId.current);

72    };

73  }, [isPlaying, bpm, grid]);

74

75  const togglePlay = () => {

76    if (!audioCtx.current) {

77      audioCtx.current = new (window.AudioContext || (window as any).webkitAudioContext)();

78    }

79    setIsPlaying(!isPlaying);

80  };

81

82  const toggleStep = (trackId: string, stepIndex: number) => {

83    setGrid((prev) => ({

84      ...prev,

85      [trackId]: prev[trackId].map((val, i) => (i === stepIndex ? !val : val)),

86    }));

87  };

88

89  return (

90    <div style={{ 

91      background: "#0a0a0a", 

92      color: "white", 

93      padding: "20px", 

94      borderRadius: "12px", 

95      fontFamily: "monospace",

96      boxShadow: "0 0 20px rgba(0,0,0,0.5)"

97    }}>

98      <div style={{ display: "flex", alignItems: "center", gap: "20px", marginBottom: "20px" }}>

99        <button 

100          onClick={togglePlay}

101          style={{

102            background: isPlaying ? "#ff0055" : "#00f2ff",

103            border: "none",

104            padding: "10px 20px",

105            borderRadius: "4px",

106            color: "black",

107            fontWeight: "bold",

108            cursor: "pointer",

109            boxShadow: `0 0 10px ${isPlaying ? "#ff0055" : "#00f2ff"}`,

110            transition: "all 0.2s"

111          }}

112        >

113          {isPlaying ? "STOP" : "PLAY"}

114        </button>

115        <div style={{ display: "flex", alignItems: "center", gap: "10px" }}>

116          <span>BPM:</span>

117          <input 

118            type="range" 

119            min="60" 

120            max="180" 

121            value={bpm} 

122            onChange={(e) => setBpm(Number(e.target.value))}

123            style={{ accentColor: "#00f2ff" }}

124          />

125          <span style={{ color: "#00f2ff" }}>{bpm}</span>

126        </div>

127      </div>

128

129      <div style={{ display: "flex", flexDirection: "column", gap: "10px" }}>

130        {TRACKS.map((track) => (

131          <div key={track.id} style={{ display: "flex", alignItems: "center", gap: "10px" }}>

132            <div style={{ width: "60px", fontSize: "12px", color: COLORS[track.id as keyof typeof COLORS] }}>

133              {track.label}

134            </div>

135            <div style={{ display: "flex", gap: "4px" }}>

136              {grid[track.id].map((active, i) => (

137                <div

138                  key={i}

139                  onClick={() => toggleStep(track.id, i)}

140                  style={{

141                    width: "20px",

142                    height: "30px",

143                    borderRadius: "2px",

144                    cursor: "pointer",

145                    background: active ? COLORS[track.id as keyof typeof COLORS] : "#1a1a1a",

146                    boxShadow: active ? `0 0 8px ${COLORS[track.id as keyof typeof COLORS]}` : "none",

147                    border: currentStep === i ? "1px solid white" : "1px solid transparent",

148                    transition: "all 0.1s"

149                  }}

150                />

151              ))}

152            </div>

153          </div>

154        ))}

155      </div>

156    </div>

157  );

158}
